func ( spanAnnotator) ( httptrace.GotConnInfo) {
	 := []trace.Attribute{
		trace.BoolAttribute("httptrace.got_connection.reused", .Reused),
		trace.BoolAttribute("httptrace.got_connection.was_idle", .WasIdle),
	}
	if .WasIdle {
		 = append(,
			trace.StringAttribute("httptrace.got_connection.idle_time", .IdleTime.String()))
	}
	.sp.Annotate(, "GotConn")
}
