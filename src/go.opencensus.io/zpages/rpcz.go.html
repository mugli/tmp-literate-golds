type statGroup struct {
	Direction string
	Snapshots []*statSnapshot
}

func ( *statGroup) () int {
	return len(.Snapshots)
}

func ( *statGroup) (,  int) {
	.Snapshots[], .Snapshots[] = .Snapshots[], .Snapshots[]
}

func ( *statGroup) (,  int) bool {
	return .Snapshots[].Method < .Snapshots[].Method
}
