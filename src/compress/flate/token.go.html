type token uint32
Convert a literal into a literal token.
func ( uint32) token { return token(literalType + ) }
Convert a < xlength, xoffset > pair into a match token.
func ( uint32,  uint32) token {
	return token(matchType + <<lengthShift + )
}
Returns the literal of a literal token
func ( token) () uint32 { return uint32( - literalType) }
Returns the extra offset of a match token
func ( token) () uint32 { return uint32() & offsetMask }

func ( token) () uint32 { return uint32(( - matchType) >> lengthShift) }

func ( uint32) uint32 { return lengthCodes[] }
