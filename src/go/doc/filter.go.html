func ( *ast.GenDecl,  Filter) bool {
	for ,  := range .Specs {
		switch v := .(type) {
		case *ast.ValueSpec:
			for ,  := range .Names {
				if (.Name) {
					return true
				}
			}
		case *ast.TypeSpec:
			if (.Name.Name) {
				return true
			}
			switch t := .Type.(type) {
			case *ast.StructType:
				if matchFields(.Fields, ) {
					return true
				}
			case *ast.InterfaceType:
				if matchFields(.Methods, ) {
					return true
				}
			}
		}
	}
	return false
}
