A fileStat is the implementation of FileInfo returned by Stat and Lstat.
type fileStat struct {
	name    string
	size    int64
	mode    FileMode
	modTime time.Time
	sys     syscall.Stat_t
}

func ( *fileStat) () int64        { return .size }
func ( *fileStat) () FileMode     { return .mode }
func ( *fileStat) () time.Time { return .modTime }
