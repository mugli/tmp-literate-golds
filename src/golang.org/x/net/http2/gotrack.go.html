var DebugGoroutines = os.Getenv("DEBUG_HTTP2_GOROUTINES") == "1"

type goroutineLock uint64

func () goroutineLock {
	if !DebugGoroutines {
		return 0
	}
	return goroutineLock(curGoroutineID())
}

func ( goroutineLock) () {
	if !DebugGoroutines {
		return
	}
	if curGoroutineID() != uint64() {
		panic("running on the wrong goroutine")
	}
}

		return
	}
	if curGoroutineID() == uint64() {
		panic("running on the wrong goroutine")

}
