All of these RemoveNonRedistributableData methods remove data that we cannot legally redistribute if the receiver is non-redistributable.

package internal

func ( *Module) () {
	for ,  := range .Licenses {
		.RemoveNonRedistributableData()
	}
	for ,  := range .Units {
		.RemoveNonRedistributableData()
	}
}

func ( *Unit) () {
	if !.IsRedistributable {
		.Readme = nil
		.Documentation = nil
	}
}

func ( *PackageMeta) () {
	if !.IsRedistributable {
		.Synopsis = ""
	}
