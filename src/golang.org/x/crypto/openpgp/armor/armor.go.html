A Block represents an OpenPGP armored structure. The encoded form is: -----BEGIN Type----- Headers base64-encoded Bytes '=' base64 encoded checksum -----END Type----- where Headers is a possibly empty sequence of Key: Value lines. Since the armored data can be very large, this package presents a streaming interface.
type Block struct {
	Type    string            // The type, taken from the preamble (i.e. "PGP SIGNATURE").
	Header  map[string]string // Optional headers.
	Body    io.Reader         // A Reader from which the contents can be read
	lReader lineReader
	oReader openpgpReader
}

type openpgpReader struct {
	lReader    *lineReader
	b64Reader  io.Reader
	currentCRC uint32
}

	return
