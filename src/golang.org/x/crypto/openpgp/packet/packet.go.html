A spanReader is an io.LimitReader, but it returns ErrUnexpectedEOF if the underlying Reader returns EOF before the limit has been reached.
type spanReader struct {
	r io.Reader
	n int64
}

func ( *spanReader) ( []byte) ( int,  error) {

	.n -= int64()
	if .n > 0 &&  == io.EOF {
		 = io.ErrUnexpectedEOF
	}
	return
}
