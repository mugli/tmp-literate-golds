truncatingMAC wraps around a hash.Hash and truncates the output digest to a given size.
type truncatingMAC struct {
	length int
	hmac   hash.Hash
}

func ( truncatingMAC) ( []byte) (int, error) {
	return .hmac.Write()
}

func ( truncatingMAC) ( []byte) []byte {
	 := .hmac.Sum()
	return [:len()+.length]
}

func ( truncatingMAC) () {
	.hmac.Reset()
}

func ( truncatingMAC) () int {
	return .length
}

func ( truncatingMAC) () int { return .hmac.BlockSize() }
