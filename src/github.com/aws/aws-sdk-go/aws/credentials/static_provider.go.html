func ( *StaticProvider) () (Value, error) {
	if .AccessKeyID == "" || .SecretAccessKey == "" {
		return Value{ProviderName: StaticProviderName}, ErrStaticCredentialsEmpty
	}

	if len(.Value.ProviderName) == 0 {
		.Value.ProviderName = StaticProviderName
	}
	return .Value, nil
}
