type Action string

const (
	Create  Action = "create"
	Update         = "update"
	Delete         = "delete"
	Invalid        = "invalid"
)

type Command struct {
	Name plumbing.ReferenceName
	Old  plumbing.Hash
	New  plumbing.Hash
}

func ( *Command) () Action {
	if .Old == plumbing.ZeroHash && .New == plumbing.ZeroHash {
		return Invalid
	}

	if .Old == plumbing.ZeroHash {
		return Create
	}

	if .New == plumbing.ZeroHash {
		return Delete
	}

	return Update
}

func ( *Command) () error {
	if .Action() == Invalid {
		return ErrMalformedCommand
	}

	return nil
