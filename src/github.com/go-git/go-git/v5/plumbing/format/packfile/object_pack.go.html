ObjectToPack is a representation of an object that is going to be into a pack file.
The main object to pack, it could be any object, including deltas
Base is the object that a delta is based on (it could be also another delta). If the main object is not a delta, Base will be null
Original is the object that we can generate applying the delta to Base, or the same object as Object in the case of a non-delta object.
Depth is the amount of deltas needed to resolve to obtain Original (delta based on delta based on ...)
	Depth int
offset in pack when object has been already written, or 0 if it has not been written yet
	Offset int64
Information from the original object
	resolvedOriginal bool
	originalType     plumbing.ObjectType
	originalSize     int64
	originalHash     plumbing.Hash
}

func ( *ObjectToPack) () plumbing.ObjectType {
	if .Original != nil {
		return .Original.Type()
	}

	if .resolvedOriginal {
		return .originalType
	}

	if .Base != nil {
		return .Base.()
	}

	if .Object != nil {
		return .Object.Type()
	}

	panic("cannot get type")
}
