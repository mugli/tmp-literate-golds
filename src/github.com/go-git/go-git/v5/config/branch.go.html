func ( *Branch) () error {
	if .Name == "" {
		return errBranchEmptyName
	}

	if .Merge != "" && !.Merge.IsBranch() {
		return errBranchInvalidMerge
	}

	if .Rebase != "" &&
		.Rebase != "true" &&
		.Rebase != "interactive" &&
		.Rebase != "false" {
		return errBranchInvalidRebase
	}

	return nil
}

func ( *Branch) () *format.Subsection {
	if .raw == nil {
		.raw = &format.Subsection{}
	}

	.raw.Name = .Name

	if .Remote == "" {
		.raw.RemoveOption(remoteSection)
	} else {
		.raw.SetOption(remoteSection, .Remote)
	}

	if .Merge == "" {
		.raw.RemoveOption(mergeKey)
	} else {
		.raw.SetOption(mergeKey, string(.Merge))
	}

	if .Rebase == "" {
		.raw.RemoveOption(rebaseKey)
	} else {
		.raw.SetOption(rebaseKey, .Rebase)
	}

	return .raw
}
