func ( *Submodule) () error {
	if .Path == "" {
		return ErrModuleEmptyPath
	}

	if .URL == "" {
		return ErrModuleEmptyURL
	}

	if dotdotPath.MatchString(.Path) {
		return ErrModuleBadPath
	}

	return nil
}

func ( *Submodule) () *format.Subsection {
	if .raw == nil {
		.raw = &format.Subsection{}
	}

	.raw.Name = .Name
	if .raw.Name == "" {
		.raw.Name = .Path
	}

	.raw.SetOption(pathKey, .Path)
	.raw.SetOption(urlKey, .URL)

	if .Branch != "" {
		.raw.SetOption(branchKey, .Branch)
	}

	return .raw
