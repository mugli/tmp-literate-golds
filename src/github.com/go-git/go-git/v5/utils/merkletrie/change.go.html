func ( *Change) () (Action, error) {
	if .From == nil && .To == nil {
		return Action(0), fmt.Errorf("malformed change: nil from and to")
	}
	if .From == nil {
		return Insert, nil
	}
	if .To == nil {
		return Delete, nil
	}

	return Modify, nil
}
