IsZero reports whether v is the zero value. This does not rely on Interface and so can be used on unexported fields.
func ( reflect.Value) bool {
	switch .Kind() {
	case reflect.Bool:
		return .Bool() == false
	case reflect.Int, reflect.Int8, reflect.Int16, reflect.Int32, reflect.Int64:
		return .Int() == 0
	case reflect.Uint, reflect.Uint8, reflect.Uint16, reflect.Uint32, reflect.Uint64, reflect.Uintptr:
		return .Uint() == 0
	case reflect.Float32, reflect.Float64:
		return math.Float64bits(.Float()) == 0
	case reflect.Complex64, reflect.Complex128:
		return math.Float64bits(real(.Complex())) == 0 && math.Float64bits(imag(.Complex())) == 0
	case reflect.String:
		return .String() == ""
	case reflect.UnsafePointer:
		return .Pointer() == 0
	case reflect.Chan, reflect.Func, reflect.Interface, reflect.Ptr, reflect.Map, reflect.Slice:
		return .IsNil()
	case reflect.Array:
		for  := 0;  < .Len(); ++ {
			if !(.Index()) {
				return false
			}
		}
		return true
	case reflect.Struct:
		for  := 0;  < .NumField(); ++ {
			if !(.Field()) {
				return false
			}
		}
		return true
	}
	return false
