Next returns true if there is another label available.
func ( *MergeItererator) () bool {
	if .one.done && .two.done {
		return false
	}
	if .one.done {
		.current = .two.label
		.two.advance()
		return true
	}
	if .two.done {
		.current = .one.label
		.one.advance()
		return true
	}
	if .one.label.Key == .two.label.Key {
		.current = .one.label // first iterator label value wins
		.one.advance()
		.two.advance()
		return true
	}
	if .one.label.Key < .two.label.Key {
		.current = .one.label
		.one.advance()
		return true
	}
	.current = .two.label
	.two.advance()
	return true
}
